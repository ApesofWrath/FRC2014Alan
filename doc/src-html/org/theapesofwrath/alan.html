246    public void test() {
247        DriverStationLCD dslcd = DriverStationLCD.getInstance();
248        Timer pause = new Timer();
249        pause.stop();
250        int[] cnt = new int[]{0, 0, 0, 0, 0};
251        Talon pwm = new Talon(1);
252        AnalogChannel ac = new AnalogChannel(1);
253        DigitalInput di = new DigitalInput(1);

260                pwm = new Talon(i);
261                pwm.set(0.3);
262                cnt[0] = i;
263            }
264            //if i is less than or equal to the number of AnalogChannels, test to see if a new instance of an AnalogChannel can be made 
265            if (i <= DriverStationLCD.kAnalogChannels) {
266                ac = new AnalogChannel(i);
267                cnt[1] = i;
268            }
269            //if i is less than or equal to the number of DigitalChannels, test to see if a new instance of a DigitalInput can be made 
270            if (i <= DriverStationLCD.kDigitalChannels) {
271                di = new DigitalInput(i);
272                cnt[2] = i;
273            }
274            //if i is less than or equal to the number of RelayChannels, test to see if a new instance of a Relay can be made 
275            if (i <= DriverStationLCD.kRelayChannels) {
276                r = new Relay(i);
277                r.set(Relay.Value.kForward);
278                cnt[3] = i;
279            }
280            //if i is less than or equal to the number of RelayChannels, test to see if a new instance of a Solenoid can be made 
281            if (i <= DriverStationLCD.kRelayChannels) {
282                s = new Solenoid(i);
283                s.set(true);
284                cnt[4] = i;
285            }
286            //print values
287            dslcd.println(DriverStationLCD.Line.kUser1, 1, "i: " + String.valueOf(i));
288            dslcd.println(DriverStationLCD.Line.kUser2, 1, "pwm " + String.valueOf(cnt[0]) + ": " + String.valueOf(pwm.get()));

295            pause.start();

297            while (pause.get() < 1.0) {
298                pause.stop();

305            ac.free();
306            di.free();
307            s.free();
308            r.free();
309            //reclaims memory of the test objects that were created
310            System.gc();

314    // handle the shifting and lifting of the shooter
315    private void pneumatics() {
316        //shift
317        if (leftStick.getRawButton(shiftUpButton) || rightStick.getRawButton(shiftUpButton)) {
318            //shift up solenoids
319            shiftUp.set(true);
320            shiftDown.set(false);
321        }//end if loop for shifting up solenoids
322        if (leftStick.getRawButton(shiftDownButton) || rightStick.getRawButton(shiftDownButton)) {
323            //shift down solenoids
324            shiftUp.set(false);
325            shiftDown.set(true);
326        }//end if loop for shifting down solenoids
327
328        // Shooter Lifter
329        if (operatorStick.getRawButton(shooterUpButton)) {
330            shooterLiftUp.set(true);
331            shooterLiftDown.set(false);
332        }//end if loop for lifting up shooter
333        if (operatorStick.getRawButton(shooterDownButton)) {
334            shooterLiftUp.set(false);
335            shooterLiftDown.set(true);
336        }//end if loop for lifting(?) down shooter 
337    }//end pneumatics

345    // State Mahine to handle all of the shooting
346    // TODO: Test, this is basically a direct port from the C++ source
347    private void shooterStateMachine() {
348        //compare entry and exit states, detect when state has changed 
349        int oldState;
350        oldState = shootingState;
351        //if (operatorStick.getRawButton(operatorStartStateMachineButton)) {
352        //    shootingState = start;
353        //}// end if loop that changes shootingState to start if operatorStartStateMachineButton is enabled
354        switch (shootingState) {
355            default:
356                shootingState = start;
357                break;
358
359            //start to idle
360            case start:
361                shootingState = idle;
362                shooterPID.setSetpoint(0);
363                // retract cylinder, just in case.
364                plunge(false);
365                break;
366
367            //from start OR stopShooterMotor to setManualShoot OR setDumpShoot
368            case idle:
369                // stop the motor when we aren't shooting
370                shooterPID.setSetpoint(0);
371                System.out.println(String.valueOf(operatorStick.getRawButton(operatorStartStateMachineButton)));
372                if (operatorStick.getRawButton(operatorStartStateMachineButton)) {
373                    shootingState = setManualShoot;
374                }
375                if (operatorStick.getRawButton(operatorDumpAStateMachineButton)
376                        || operatorStick.getRawButton(operatorDumpBStateMachineButton)) {
377                    shootingState = setDumpShoot;
378                }
379                break;
380
381            //from idle to startShooterMotor
382            case setManualShoot:
383                shootingMode = manualShoot;
384                shootingState = startShooterMotor;
385                //System.out.println(String.valueOf(MAX_RPM * (-(1 - operatorStick.getThrottle()) / 2)));
386                shooterPID.setSetpoint(MAX_RPM * ((1 + operatorStick.getZ()) / 2));
387                break;
388
389            //fom idle to startShooterMotor
390            case setDumpShoot:
391                shootingMode = dump;
392                shootingState = startShooterMotor;
393                break;
394
395            //from startShooterMotor OR retractCylinderWait to extendCylinder
396            case shooterCheckTrigger:
397                //buttton one is trigger
398                shooterPID.setSetpoint(MAX_RPM * ((1 + operatorStick.getZ()) / 2));
399                if (operatorStick.getRawButton(1)) {
400                    shootingState = extendCylinder;
401                }
402                /*else {
403                 shootingState = idle;
404                 }*/
405                break;
406            //from setManualShoot OR setDumpShoot to shooterMotorSpinupWait OR shooterCheckTrigger
407            case startShooterMotor:
408                shooterPID.setSetpoint(DUMP_RPM);
409                if (shootingMode == dump) {
410                    shooterTimer.reset();
411                    shootingState = shooterMotorSpinupWait;
412                }//end if (dump)
413                                /* for manual shooting, the operator has to set the shooter motor speed manually. 
414                 It is expected that the operator will wait for spinup before pulling the trigger*/
415                if (shootingMode == manualShoot) {
416                    shootingState = shooterCheckTrigger;
417                }
418
419                break;
420
421            //from startShooterMotor to extendCylinder
422            case shooterMotorSpinupWait:
423                if (shooterTimer.get() >= SHOOTER_MOTOR_SPINUP_WAIT) {
424                    shootingState = extendCylinder;
425                }
426                break;
427
428            //from shooterMotorSpinupWait OR shooterCheckTrigger OR checkFrisbeeCount to extendCylinderWait 
429            case extendCylinder:
430                //cout<<"firing"<<endl;
431                plunge(true);
432                shooterTimer.reset();
433                shootingState = extendCylinderWait;
434                break;
435
436            //from extendCylinder to retractCylinder 
437            case extendCylinderWait:
438                if (shooterTimer.get() >= SHOOTER_CYLINDER_EXTEND_TIME) {
439                    shooterTimer.reset();
440                    shootingState = retractCylinder;
441                }
442                break;
443
444            //from extendCylinderWait to retractCylinderWait 
445            case retractCylinder:
446                plunge(false);
447                shooterTimer.reset();
448                shootingState = retractCylinderWait;
449                break;
450
451            //from retractCylinder to checkFrisbeCount OR shooterCheckTrigger
452            case retractCylinderWait:
453                if (shooterTimer.get() >= SHOOTER_CYLINDER_RETRACT_TIME) {
454                    shooterTimer.reset();
455                    if (shootingMode == dump) {
456                        shootingState = checkFrisbeeCount;
457                    }
458                    if (shootingMode == manualShoot) {
459                        shootingState = shooterCheckTrigger;
460                    }
461                }
462                break;
463
464            //from retractCylinderWait to stopShooterMotor OR extendCylinder 
465            case checkFrisbeeCount:
466                --remainingFrisbees;
467                //number of frisbees decrement when?
468                if (remainingFrisbees <= 0) {
469                    remainingFrisbees = 4;
470                    shootingState = stopShooterMotor;
471                } else {
472                    shootingState = extendCylinder;
473                }
474                break;
475
476            //from checkFrisbeeCount to idle                        
477            case stopShooterMotor:
478                shooterPID.setSetpoint(0);
479                shootingState = idle;
480                break;
481        }//end Switch(shootingState)
482    }// end shooterController
